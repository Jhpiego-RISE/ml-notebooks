def one_hot_encode_col(df,dummies_col:str,prefix_entry:str):
    if df is not None:
        try:
            dummies = pd.get_dummies(df[dummies_col], prefix=prefix_entry,drop_first=False)
            df = df.join(dummies)
            return df
        except Exception as e:
            print(f"Exception on one hot encoding with error: {e}")

def evaluate_gender_specific(test, y_test, predictions, probs):

    y_test_female_only = y_test[condition_is_female]

    # Filter the corresponding predicted target results: y_predicted
    y_predicted_male_only = predictions[condition_is_male]
    y_predicted_female_only = predictions[condition_is_female]

    # Filter the corresponding predicted target results: y_predicted
    y_probs_male_only = probs[condition_is_male]

    # Let's now calculate these metrics for ONLY THE MALES
    tree_accuracy_males_only = accuracy_score(y_test_male_only, y_predicted_male_only)*100
